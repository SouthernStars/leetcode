位运算技巧
- 消去最后一位的1: 使用 x & (x-1) 可以消去整数 x 的二进制表示中最右边的1
- 检测2的幂次: 利用 x & (x-1) 的性质，可以检测一个数是否为2的幂。如果 x 是2的幂，则 x > 0 且 x 的二进制表示中只有一个1，因此 x & (x-1) 应该返回0
- 计算二进制中1的个数: 通过不断使用 x & (x-1) 操作消去1，当 x 变为0时，操作的次数即为二进制中1的个数
- 计算数组中唯一出现的次数：异或操作 x ^ y 具有以下性质：x ^ x = 0 和 x ^ 0 = x。这可以用来在数组中找出唯一出现一次的数，因为其他数都出现两次，它们的异或结果为0
- 交换两个数: 使用异或操作 a ^= b; b ^= a; a ^= b; 可以在不使用额外空间的情况下交换两个数
- 检测整数的奇偶性: 通过 n & 1 可以检测整数 n 的奇偶性。如果结果为0，则 n 是偶数；如果结果为1，则 n 是奇数
- 获取最后一位：n & 1


题目描述：给定两个无符号4字节长度整型x和y,将x和y经过zorder计算后得出一个无符号8字节长度整型值z 。
zorder的计算规则如下:将x和y转成二进制z,二进制数值的每两位分别从x和y中取 每次获取，x作为二进制的高位，y作为低位直到x和 y的 32 位取完，即得出 z值

实例：如:x是 18，二进制00010010。y是52 ,二进制 00110100。z为 1816 二进制为  011100011000

```go
package main

import (
	"fmt"
)
func main() {
    // 处理输入
    var n int 
    fmt.Scan(&n)
    var x, y uint32
    var z uint64
    for i:=1; i<=n; i++ {
        fmt.Scan(&x, &y)
        z = 0
        for j:=0; j<32; j++ {
            // 获取低位
            yh := (y >> j) & 1
            if yh != 0 {
                z |= uint64(1) << (2*j)
            }
            // 获取高位
            xh := (x >> j) & 1 
            if xh != 0 {
                z |= uint64(1) << (2*j+1)
            }
        }
        fmt.Println(z)
    }
}
package main
import (
"fmt"
)
func main() {
	// 处理输入
	var n, x, y int
	fmt.Scan(&n)
	for i:=1; i<=n; i++ {
		fmt.Scan(&x, &y)
		ans := deal(x, y)
		fmt.Println(ans)
	}
}
func deal(x, y int) int {
	xb := tobitarry(x)
	yb := tobitarry(y)
	zb := [64]int{}
	for i:=0; i<=31; i++ {
		zb[2*i] = xb[i]
		zb[2*i+1] = yb[i]
	}
	// 求和 
	p := 1
	ans := 0
	for i:=63; i>=0; i-- {
		if zb[i] == 1 {
			ans += p
		}
		p = p*2
	}
	return ans
}
func tobitarry(x int) (ans [32]int) {
	idx := 31
	for x != 0 {
		ans[idx] = x % 2
		x = x / 2
		idx --
	}
	return ans
}
```
