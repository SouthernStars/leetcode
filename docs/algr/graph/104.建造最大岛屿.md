# 104.沉没孤岛
[卡玛网 链接](https://kamacoder.com/problempage.php?pid=1176)

### 一、题目描述

题目描述：

给定一个由 1（陆地）和 0（水）组成的矩阵，你最多可以将矩阵中的一格水变为一块陆地，在执行了此操作之后，矩阵中最大的岛屿面积是多少。

岛屿面积的计算方式为组成岛屿的陆地的总数。岛屿是被水包围，并且通过水平方向或垂直方向上相邻的陆地连接而成的。你可以假设矩阵外均被水包围。

输入描述：

第一行包含两个整数 N, M，表示矩阵的行数和列数。之后 N 行，每行包含 M 个数字，数字为 1 或者 0，表示岛屿的单元格。

输出描述：

输出一个整数，表示最大的岛屿面积。

### 二、解题思路

深度优先遍历: 使用深度优先遍历为每个陆地贴编号，比如2，3（修改grid），使用哈希表记录每个陆地的编号及对应的陆地数

然后再两层遍历grid 中的水域，假设变成陆地，查看上下左右邻接那些编号的陆地，这样陆地面积就得到了，在遍历过程中记录最大面积


### 三、代码实现

```go
package main
import "fmt"
// 上下左右
var directions = [4][2]int{{-1, 0}, {1, 0}, {0, -1}, {0, 1}}

func main() {
	var n, m int
	fmt.Scanf("%d %d", &n, &m)
	grid := make([][]int, n)
	vist := make([][]bool, n)
	for i := 0; i < n; i++ {
		grid[i] = make([]int, m)
		for j := 0; j < m; j++ {
			fmt.Scanf("%d", &grid[i][j])
		}
		vist[i] = make([]bool, m)
	}
	// 使用 map 记录
	markCnt := map[int]int{}
	mark := 1
	maxCnt := 0
	for i := 0; i < n; i++ {
		for j := 0; j < m; j++ {
			if !vist[i][j] && grid[i][j] == 1 {
				mark++
				count := 0
				dfs(&grid, &vist, i, j, mark, &count)
				markCnt[mark] = count
				maxCnt = max(maxCnt, count)
			}
		}
	}
	// 双层遍历
	for i := 0; i < n; i++ {
		for j := 0; j < m; j++ {
			if grid[i][j] == 0 {
				cnt := 1
				nerb := map[int]bool{}
				for _, dir := range directions {
					newX, newY := i+dir[0], j+dir[1]
					if newX >= 0 && newX < len(grid) && newY >= 0 && newY < len(grid[0]) && grid[newX][newY] != 0 {
						nerb[grid[newX][newY]] = true
					}
				}
				for k, _ := range nerb {
					cnt += markCnt[k]
				}
				maxCnt = max(maxCnt, cnt)
			}
		}
	}
	fmt.Println(maxCnt)
}

func dfs(grid *[][]int, vist *[][]bool, x, y, mark int, count *int) {
	(*vist)[x][y] = true
	(*grid)[x][y] = mark
	*count++
	for _, dir := range directions {
		newX, newY := x+dir[0], y+dir[1]
		if newX >= 0 && newX < len(*grid) && newY >= 0 && newY < len((*grid)[0]) {
			if (*grid)[newX][newY] == 1 && !(*vist)[newX][newY] {
				dfs(grid, vist, newX, newY, mark, count)
			}
		}
	}
}
```
