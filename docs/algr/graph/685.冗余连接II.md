# 685.冗余连接II
[leetcode 链接](https://leetcode.cn/problems/redundant-connection-ii/description/)

### 一、题目描述

在本问题中，有根树指满足以下条件的 有向 图。该树只有一个根节点，所有其他节点都是该根节点的后继。该树除了根节点之外的每一个节点都有且只有一个父节点，而根节点没有父节点。

输入一个有向图，该图由一个有着 n 个节点（节点值不重复，从 1 到 n）的树及一条附加的有向边构成。附加的边包含在 1 到 n 中的两个不同顶点间，这条附加的边不属于树中已存在的边。

结果图是一个以边组成的二维数组 edges 。 每个元素是一对 [ui, vi]，用以表示 有向 图中连接顶点 ui 和顶点 vi 的边，其中 ui 是 vi 的一个父节点。

返回一条能删除的边，使得剩下的图是有 n 个节点的有根树。若有多个答案，返回最后出现在给定二维数组的答案。

并查集


### 三、代码实现

var father []int

func findRedundantDirectedConnection(edges [][]int) (ans []int) {
    N := len(edges)
	father = make([]int, N+1)
	indegree := make([]int, N+1)
    
    // 统计每个结点的入度
	for i := 0; i < N; i++ {
		indegree[edges[i][1]]++
	}
	// 统计入度为2 的点及关联的边，按照逆序
	var twoEdges []int
	for i := N - 1; i >= 0; i-- {
		if indegree[edges[i][1]] == 2 {
			twoEdges = append(twoEdges, i)
		}
	}

    if len(twoEdges) == 0 { // 存在环
		Init(N)
		for i := 0; i < N; i++ {
			if IsSameRoot(edges[i][0], edges[i][1]) {
                ans = []int{ edges[i][0], edges[i][1]}
				return
			} else {
				Join(edges[i][0], edges[i][1])
			}
		}
	} 
    // 存在入度为 2 的点
    deleteEdge := twoEdges[0]
    Init(N)
    for i := 0; i < N; i++ {
        if i == deleteEdge {
            continue
        }
        if IsSameRoot(edges[i][0], edges[i][1]) { // 删除边后，构成了环，因此删错了, 修改删除边
            deleteEdge = twoEdges[1]
            break
        } else {
            Join(edges[i][0], edges[i][1])
        }
    }
    ans = []int{edges[deleteEdge][0], edges[deleteEdge][1]}
    return
}


// Init 初始化并查集，编号1~n
func Init(n int) {
	for i := 1; i <= n; i++ {
		father[i] = i
	}
}

// FindRoot 寻找 u 的根
func FindRoot(u int) int {
	if u == father[u] {
		return u
	}
	father[u] = FindRoot(father[u])
	return father[u]
}

func Join(u, v int) {
	u = FindRoot(u)
	v = FindRoot(v)
	if u != v {
		father[v] = u
	}
}

func IsSameRoot(u, v int) bool {
	return FindRoot(u) == FindRoot(v)
}
