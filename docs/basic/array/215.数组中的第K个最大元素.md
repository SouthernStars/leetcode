# 215.数组中的第K个最大元素
[leetcode 链接](https://leetcode.cn/problems/kth-largest-element-in-an-array/description/)

### 一、题目描述

给定整数数组 nums 和整数 k，请返回数组中第 k 个最大的元素。

请注意，你需要找的是数组排序后的第 k 个最大的元素，而不是第 k 个不同的元素。

你必须设计并实现时间复杂度为 O(n) 的算法解决此问题。

### 二、解题思路

// 使用基于快排序的方法寻找，每次划分可以确定一个元素所在的位置
// 假设从小到大排序 ...... x .....
// 根据改位置和 k ，继续在左侧或则右侧继续划分

### 三、代码实现

```c++

int quickselect(vector<int>& nums, int l, int r, int k) {
    if (l == r) {
        return nums[k];
    }
    int patition = nums[l];
    int i = l - 1, j = r + 1;
    while (i < j) {
        do
        i++;
        while (nums[i] < patition);
        do
        j--;
        while (nums[j] > patition);
        if (i < j) {
            swap(nums[i], nums[j]);
        }
    }
    if (k <= j) {
        return quickselect(nums, l, j, k);
    } else {
        return quickselect(nums, j + 1, r, k);
    }
}
int findKthLargest(vector<int>& nums, int k) {
    return quickselect(nums, 0, nums.size() - 1, nums.size() - k);
}
```

```c++
void maxHeapify(vector<int>& nums, int cur, int heapSize) {
    int l = 2 * cur + 1, r = 2 * cur + 2;
    int large = cur;
    if (l < heapSize && nums[l] > nums[large]) {
        large = l;
    }
    if (r < heapSize && nums[r] > nums[large]) {
        large = r;
    }
    if (large != cur) {
        swap(nums[large], nums[cur]);
        maxHeapify(nums, large, heapSize);
    }
}
void buildmaxHeap(vector<int>& nums, int heapSize) {
    for (int i = heapSize / 2 - 1; i >= 0; i--) {
        maxHeapify(nums, i, heapSize);
    }
}
int findKthLargest(vector<int>& nums, int k) {
    int heapSize = nums.size();
    buildmaxHeap(nums, heapSize);
    for (int i = 0; i < k - 1; i++) {
        swap(nums[0], nums[heapSize - 1]);
        maxHeapify(nums, 0, --heapSize);
    }
    return nums[0];
}
```
