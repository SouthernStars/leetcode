# 215.数组中的第K个最大元素
[leetcode 链接](https://leetcode.cn/problems/kth-largest-element-in-an-array/description/)

### 一、题目描述

给定整数数组 nums 和整数 k，请返回数组中第 k 个最大的元素。

请注意，你需要找的是数组排序后的第 k 个最大的元素，而不是第 k 个不同的元素。

你必须设计并实现时间复杂度为 O(n) 的算法解决此问题。

### 二、解题思路

// 使用基于快排序的方法寻找，每次划分可以确定一个元素所在的位置
// 假设从小到大排序 ...... x .....
// 根据改位置和 k ，继续在左侧或则右侧继续划分

### 三、代码实现

```go

// 堆排序
type maxHeap []int 
// 需要实现结构 Len/Less/Swap/Push/Pop
func (hp maxHeap) Len() int {
    return len(hp)
}
func (hp maxHeap) Less(i, j int) bool {
    return hp[i] > hp[j]
}
func (hp maxHeap) Swap(i, j int) {
    hp[i], hp[j] = hp[j], hp[i]
}
func (hp *maxHeap) Push(x interface{}) {
    *hp = append(*hp, x.(int))
}
func (hp *maxHeap) Pop() interface{} {
    x := (*hp)[len(*hp)-1]
    *hp = (*hp)[0:len(*hp)-1]
    return x
}
func findKthLargest(nums []int, k int) int {
    hp := &maxHeap{}
    heap.Init(hp)
    for i:=0; i<len(nums); i++ {
        heap.Push(hp, nums[i])
    }

    // 寻找第k个
    var ans int
    for i:=0; i<k; i++ {
        ans = heap.Pop(hp).(int)
    }
    return ans
}
```

```go
func findKthLargest(nums []int, k int) int {
    n := len(nums)
    sort.Ints(nums)
    return nums[n-k]
}

func findKthLargest(nums []int, k int) int {
    n := len(nums)
    return quickselect(nums, 0, n-1, n-k)
}

func quickselect(nums []int, left int, right int, pos int) int {
    if left == right {
        return nums[pos]
    }
    i, j, pivot := left, right, left
    for i < j {
        for i<j&&nums[j]>=nums[pivot] {
            j--
        }
        for i<j&&nums[i]<=nums[pivot] {
            i++
        }
        if i < j {
            nums[i], nums[j] = nums[j], nums[i]
        }
    }
    // j 位置即为 pivot 的位置
    nums[j], nums[pivot] = nums[pivot], nums[j]
    if j<pos {
        return quickselect(nums, j+1, right, pos)
    } else {
        return quickselect(nums, left, j, pos)
    }
}
```
