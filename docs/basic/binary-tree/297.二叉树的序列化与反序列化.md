# 297.二叉树的序列化与反序列化
[leetcode 链接](https://leetcode.cn/problems/serialize-and-deserialize-binary-tree/description/)

### 一、题目描述

序列化是将一个数据结构或者对象转换为连续的比特位的操作，进而可以将转换后的数据存储在一个文件或者内存中，同时也可以通过网络传输到另一个计算机环境，采取相反方式重构得到原数据。

请设计一个算法来实现二叉树的序列化与反序列化。这里不限定你的序列 / 反序列化算法执行逻辑，你只需要保证一个二叉树可以被序列化为一个字符串并且将这个字符串反序列化为原始的树结构。

提示: 输入输出格式与 LeetCode 目前使用的方式一致，详情请参阅 LeetCode 序列化二叉树的格式。你并非必须采取这种方式，你也可以采用其他的方法解决这个问题。

### 二、解题思路

我们可以先序遍历这颗二叉树，遇到空子树的时候序列化成 None，否则继续递归序列化。那么我们如何反序列化呢？

如果当前的元素为 None，则当前为空树
否则先解析这棵树的左子树，再解析它的右子树

// Serializes a tree to a single string.
func (this *Codec) serialize(root *TreeNode) string {
    sb := &strings.Builder{}

    var pre func(*TreeNode)
    pre = func(node *TreeNode) {
        if node == nil {
            sb.WriteString("null,")
            return 
        }
        digit := strconv.Itoa(node.Val)
        sb.WriteString(digit)
        sb.WriteString(",")
        pre(node.Left)
        pre(node.Right)
    }
    pre(root)
    return sb.String()
}

// Deserializes your encoded data to tree.
func (this *Codec) deserialize(data string) *TreeNode {    
   sp := strings.Split(data, ",")

   var build func() *TreeNode 
   build = func() *TreeNode  {
        if sp[0] == "null" {
            sp = sp[1:]
            return nil
        }
        root := &TreeNode{}
        dig,_ := strconv.Atoi(sp[0])
        root.Val = dig
        sp = sp[1:]
        root.Left = build()
        root.Right = build()
        return root
   } 
   return build()
}
