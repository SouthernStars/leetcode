# 23.合并K个升序链表
[leetcode 链接](https://leetcode.cn/problems/merge-k-sorted-lists/description/)

### 一、题目描述

给你一个链表数组，每个链表都已经按升序排列。 请你将所有链表合并到一个升序链表中，返回合并后的链表。

### 二、解题思路

思路：顺序合并 k 个链表, 
优化方法：分治合并


### 三、代码实现

```go
// 顺序合并
func mergeKLists(lists []*ListNode) *ListNode {
    var ans *ListNode
    for i:=0; i<len(lists); i++ {
        ans = mergeList(ans, lists[i])
    }
    return ans
}

// 将
func mergeList(list1 *ListNode, list2 *ListNode) *ListNode {
    // 假头结点
    dummy := &ListNode{0, nil}
    tail := dummy
    for list1 != nil && list2 != nil {
        if list1.Val <= list2.Val { // 拼接 list1
            tail.Next = list1
            tail = tail.Next
            list1 = list1.Next
        } else {
            tail.Next = list2
            tail = tail.Next
            list2 = list2.Next
        }
    }
    if list1 != nil {
        tail.Next= list1
    }
    if list2 != nil {
        tail.Next = list2
    }
    return dummy.Next
}
```

```go
 // 分治合并 k 个链表
func mergeKLists(lists []*ListNode) *ListNode { 
    return merge(lists, 0, len(lists)-1)
}


func merge(lists []*ListNode, low int, high int) *ListNode {
    if low > high {
        return nil
    }
    if low == high {
        return lists[low]
    }
    mid := (low + high) / 2
    return mergeList(merge(lists, low, mid), merge(lists, mid+1, high))
}


func mergeList(list1 *ListNode, list2 *ListNode) *ListNode {

    dummy := &ListNode{0, nil}
    tail := dummy

            tail.Next = list2
            tail = tail.Next
            list2 = list2.Next
        }
    }
    if list1 != nil {
        tail.Next= list1
    }
    if list2 != nil {
        tail.Next = list2
    }
    return dummy.Next
}
```
