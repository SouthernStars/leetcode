# 85.最大矩形
[leetcode 链接](https://leetcode.cn/problems/maximal-rectangle/description/)

### 一、题目描述
给定一个仅包含 0 和 1 、大小为 rows x cols 的二维二进制矩阵，找出只包含 1 的最大矩形，并返回其面积。


### 二、解题思路

单调栈：转化成 84 题，然后遍历柱状图，得出全局最大值

### 三、代码实现


```go
func maximalRectangle(matrix [][]byte) int {
    // m 行 n 列
    m, n := len(matrix), len(matrix[0])

    graph := make([][]int, m)
    for i:=0; i<m; i++ {
        one := make([]int, n)
        for j:=0; j<n; j++ {
            if i==0 {
                if matrix[i][j] == '1' {
                    one[j] = 1
                }
            } else {
                if matrix[i][j] == '0' {
                    one[j] = 0
                } else {
                    one[j] = 1+graph[i-1][j]
                }
            }
        }
        graph[i] = one
    }
    // 对于每一行 0~m 应用单调栈
    maxArea := 0
    for i:=0; i<m; i++ {
        // 左栈记录第一个比它小的元素
        lstack := []int{}
        lhashtable := map[int]int{}
        for j:=0; j<n; j++ {
            for len(lstack) > 0 && graph[i][lstack[len(lstack)-1]] >= graph[i][j] {
                lstack = lstack[0:len(lstack)-1]
            }
            if len(lstack)==0 {
                lhashtable[j] = -1
            } else {
                lhashtable[j] = lstack[len(lstack)-1]
            }
            lstack = append(lstack, j)
        }
        // 右栈记录第一个比它小的元素
        rstack := []int{}
        rhashtable := map[int]int{}
        for j:=n-1; j>=0; j-- {
            for len(rstack) > 0 && graph[i][rstack[len(rstack)-1]] >= graph[i][j] {
                rstack = rstack[0:len(rstack)-1]
            }
            if len(rstack)==0 {
                rhashtable[j] = n
            } else {
                rhashtable[j] = rstack[len(rstack)-1]
            }
            rstack = append(rstack, j)
        }
        // 枚举中间柱子，计算面积
        for j:=0; j<n; j++ {
            height := graph[i][j]
            lindex := lhashtable[j]+1
            rindex := rhashtable[j]-1
            area := height*(rindex-lindex+1)
            maxArea = max(maxArea, area)
        }
    }
    return maxArea
}
```
